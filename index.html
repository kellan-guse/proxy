const { app, BrowserWindow } = require('electron'); function createWindow() { const win = new BrowserWindow({ width: 1200, height: 800, webPreferences: { nodeIntegration: false, contextIsolation: true } }); // Load our custom UI win.loadFile('index.html'); } app.disableHardwareAcceleration(); // Mandatory for cloud hosting app.whenReady().then(createWindow); app.on('window-all-closed', () => { if (process.platform !== 'darwin') app.quit(); });
